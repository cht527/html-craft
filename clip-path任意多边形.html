function polygon(n=4,r=50) {
			var deg = 2 * Math.PI / n;
			var arr = [];
			for (var i=1;i<=n;i++) {
				var degree= deg*i;
				var x = r + r*Math.cos(degree) + '%';
				var y = r + r*Math.sin(degree) + '%';
				arr.push(x+ ' ' +y);
			}
			return `polygon(${arr.join(',')})`
		}
		cc.style.clipPath=polygon(6);
